192 friend inline std::ostream& operator<<(std::ostream& os, const RequestData &r) {
193 if (r.eException != SIMCONNECT_EXCEPTION_NONE)
194 os << "SimConnect exception in packet ID " << r.dwSendId << ": " << exceptionName(r.eException) << " for request: ";
195 if (r.sMethod.empty())
196 os << "Request record not found.";
197 else
198 os << r.sMethod << '(' << r.ssArguments.str() << ')';
199 if (r.dwExceptionIndex && r.dwExceptionIndex <= r.argsCount)
200 os << " (error @ arg# " << r.dwExceptionIndex << ')';
201 else if ((int)r.dwExceptionIndex == -1)
202 os << " (not due to args)";
203 return os;
204 }

233 void setMaxRecords(uint32_t maxRecords)
234 {
235 if (maxRecords != m_maxRecords) {
236 m_dataResizing = true;
237 m_data.resize(maxRecords /*, RequestData(-1)*/);
238 if (maxRecords < m_maxRecords) {
239 m_data.shrink_to_fit();
240 m_dataIndex = maxRecords ? std::min<int>(m_dataIndex, maxRecords - 1) : 0;
241 }
242 m_maxRecords = maxRecords;
243 m_dataResizing = false;
244 }
245 }

251 template <typename... Args>
252 void addRequestRecord(HANDLE hSim, std::string &&methodInfo, Args... args) {
253 if (m_dataResizing || !m_maxRecords)
254 return;
255 DWORD dwSendId = 0;
256 if (SimConnect_GetLastSentPacketID(hSim, &dwSendId) == 0) {
257 std::stringstream ss;
258 streamArgs(ss, args...);
259 m_dataIndex = (m_dataIndex + 1) % m_maxRecords; // atomic (if SCRT_THREADSAFE), guaranteed to be accessing a unique vector index.
260 m_data[m_dataIndex] = RequestData(dwSendId, std::move(methodInfo), std::move(ss), (uint8_t)sizeof...(args));
261 }
262 }

275 RequestData const & getRequestRecord(uint32_t dwSendId, /*SIMCONNECT_EXCEPTION*/ uint32_t ex = SIMCONNECT_EXCEPTION_NONE, uint32_t idx = 0)
276 {
277 static RequestData nullReq{ -1 };
278 RequestData *d = nullptr;
279 if (!m_dataResizing && m_maxRecords) {
280 // start at 10 records back from current index since it is more likely that the sendId error relates to a recent request than an old one
281 int i = (m_maxRecords + m_dataIndex - std::min<uint32_t>(10UL, m_maxRecords)) % m_maxRecords,
282 e = i;
283 for (bool first = true; i != e || first; i = (i + 1) % m_maxRecords, first = false) {
284 if (m_data[i].dwSendId == dwSendId) {
285 d = &m_data.at(i);
286 break;
287 }
288 }
289 }
290 if (!d) {
291 d = &nullReq;
292 d->dwSendId = dwSendId;
293 }
294 d->eException = (SIMCONNECT_EXCEPTION)ex;
295 d->dwExceptionIndex = idx;
296 return *d;
297 }
